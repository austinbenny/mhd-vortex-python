\section{Introduction and Problem Motivation}

\begin{frame}{Why MHD}
  \begin{itemize}
    \item Conducting fluid in motion = distributed current; \emph{Lorentz force} $\mathbf{J}\times\mathbf{B}$ acts on every cell~\cite{chen2016,freidberg2014}.
    \item \emph{Frozen-in flux}: in the ideal limit, field lines move with the fluid.
    \item Applies wherever plasma is collisional and magnetic Reynolds is large: solar wind, tokamaks, astrophysical jets, magnetospheres.
    \item New physics vs Euler:
      \begin{itemize}\small
        \item \emph{Alfven} waves---transverse, like waves on a tense string.
        \item \emph{Fast/slow magnetosonic}---sound waves modified by magnetic pressure.
        \item Maxwell stresses $-B_iB_j$ in the momentum flux.
        \item No-monopoles constraint $\nabla\cdot B = 0$.
      \end{itemize}
  \end{itemize}
  \vspace{0.3em}
  \textbf{Project goal}: 2D ideal MHD finite-volume solver in Python, validated on Orszag-Tang.
\end{frame}

\begin{frame}{Why Orszag-Tang}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \begin{itemize}
        \item Canonical 2D MHD benchmark.
        \item All-sinusoidal ICs on $[0,1]^{2}$, periodic on every side.
        \item No geometry, no density jump, no outflow BCs.
        \item If the result is wrong, it is the solver, not the setup.
        \item Reference: \cite{londrillo2000,athena_orszag_tang}.
      \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
      \centering
      \includegraphics[width=\linewidth]{rho_final.pdf} \\
      \scriptsize density $\rho(x,y,\,t = 0.5)$
    \end{column}
  \end{columns}
\end{frame}

\section{Mathematical Model}

\begin{frame}{Ideal MHD in conservative form}
  \[
  \frac{\partial U}{\partial t} + \frac{\partial F(U)}{\partial x} + \frac{\partial G(U)}{\partial y} = 0,
  \qquad
  U = (\rho,\; \rho u,\; \rho v,\; \rho w,\; B_x,\; B_y,\; B_z,\; E)^{T}.
  \]
  Closure (ideal gas, $\gamma = 5/3$):
  \[
  E = \frac{p}{\gamma - 1} + \tfrac{1}{2}\rho|\mathbf{v}|^{2} + \tfrac{1}{2}|\mathbf{B}|^{2},
  \qquad
  p_T = p + \tfrac{1}{2}|\mathbf{B}|^{2}.
  \]
  \begin{itemize}
    \item $\rho$: mass density. $\mathbf{v} = (u,v,w)$: velocity. $\mathbf{B} = (B_x,B_y,B_z)$: magnetic field.
    \item $E$: total energy density (internal $+$ kinetic $+$ magnetic).
    \item $p$: gas pressure. $\tfrac{1}{2}|\mathbf{B}|^{2}$: \emph{magnetic pressure}. $p_T$: total (gas $+$ magnetic).
    \item All 3 velocity and field components carried even in 2D so out-of-plane Alfven waves resolve correctly~\cite{powell1999}.
    \item Seven-wave fan: 2 fast, 2 Alfven, 2 slow, 1 entropy~\cite{brio1988,miyoshi2005}.
  \end{itemize}
\end{frame}

\begin{frame}{$x$-direction flux $F(U)$}
  \centering
  \footnotesize
  \begin{tabular}{c@{\quad}l@{\qquad}l}
  \toprule
  Row & $F(U)$ component & Physical meaning \\
  \midrule
  1 & $\rho u$                                       & mass flux (Euler) \\
  2 & $\rho u^{2} + p_T - B_x^{2}$                   & $x$-momentum: total pressure $p_T$, tension $-B_x^{2}$ \\
  3 & $\rho u v - B_x B_y$                           & Maxwell shear stress \\
  4 & $\rho u w - B_x B_z$                           & Maxwell shear stress \\
  5 & $0$                                            & $B_x$ self-flux $= 0$ in ideal MHD; GLM fills it \\
  6 & $u B_y - v B_x$                                & induction-equation flux for $B_y$ \\
  7 & $u B_z - w B_x$                                & induction-equation flux for $B_z$ \\
  8 & $(E + p_T)\,u - B_x(\mathbf{v}\!\cdot\!\mathbf{B})$ & enthalpy carry $+$ Poynting flux \\
  \bottomrule
  \end{tabular}
  \vspace{0.6em}
  \begin{itemize}\scriptsize
    \item Rows 6--7 are components of the motional EMF $\mathbf{v}\!\times\!\mathbf{B}$ that drag field lines with the fluid (frozen-in flux).
    \item Row 8 carries Euler enthalpy $(E + p_T)u$ \emph{plus} the Poynting flux $-B_x(\mathbf{v}\!\cdot\!\mathbf{B})$, which transports magnetic energy with the fluid.
  \end{itemize}
\end{frame}

\begin{frame}{$y$-direction flux $G(U)$}
  \centering
  \footnotesize
  \begin{tabular}{c@{\quad}l@{\qquad}l}
  \toprule
  Row & $G(U)$ component & Physical meaning \\
  \midrule
  1 & $\rho v$                                       & mass flux (Euler) \\
  2 & $\rho v u - B_y B_x$                           & Maxwell shear stress \\
  3 & $\rho v^{2} + p_T - B_y^{2}$                   & $y$-momentum: total pressure $p_T$, tension $-B_y^{2}$ \\
  4 & $\rho v w - B_y B_z$                           & Maxwell shear stress \\
  5 & $v B_x - u B_y$                                & induction-equation flux for $B_x$ \\
  6 & $0$                                            & $B_y$ self-flux $= 0$ in ideal MHD; GLM fills it \\
  7 & $v B_z - w B_y$                                & induction-equation flux for $B_z$ \\
  8 & $(E + p_T)\,v - B_y(\mathbf{v}\!\cdot\!\mathbf{B})$ & enthalpy carry $+$ Poynting flux \\
  \bottomrule
  \end{tabular}
  \vspace{0.6em}
  \begin{itemize}\scriptsize
    \item Same structure as $F(U)$ with $(u, B_x)$ and $(v, B_y)$ swapped.
    \item Implementation reuses the $x$-direction Riemann solver after the coordinate swap; halves the code surface.
  \end{itemize}
\end{frame}

\begin{frame}{The divergence constraint}
  \begin{itemize}
    \item $\nabla\cdot B = 0$ is Gauss's law: no monopoles $\Rightarrow$ no field-line sources.
    \item Exact MHD preserves this; discrete updates leave a small divergence at every cell~\cite{balsara1999,powell1999}.
    \item Why it matters: rewrite the Lorentz force as
      \[
      (\nabla\times B)\times B = (B\cdot\nabla)B - \tfrac{1}{2}\nabla|B|^{2} - (\nabla\cdot B)B.
      \]
      The last term vanishes analytically; once $\nabla\cdot B$ drifts, it acts as a spurious force parallel to $B$ and the run blows up.
    \item Fix here: \textbf{GLM} cleaning~\cite{dedner2002}---one scalar $\psi$, drops into unstaggered FVM. (CT, Powell are alternatives.)
  \end{itemize}
\end{frame}

\section{Numerical Model}

\begin{frame}{Computational mesh}
  \begin{itemize}
    \item Uniform $128\times 128$ Cartesian grid, $\Delta x = \Delta y = 1/128$.
    \item Two ghost layers per side (periodic fills for Orszag-Tang).
    \item ICs (Londrillo \& Del Zanna~\cite{londrillo2000}): $\rho = 1$, $p = 1/\gamma$, $B_0 = 1/\gamma$, and
      \begin{align*}
      \mathbf{v}(x,y,0) &= (-\sin 2\pi y,\,\sin 2\pi x,\,0), \\
      \mathbf{B}(x,y,0) &= B_0\,(-\sin 2\pi y,\,\sin 4\pi x,\,0).
      \end{align*}
    \item Left panel: $\mathbf{v}$ overlay. Right panel: $\mathbf{B}$ overlay.
  \end{itemize}
\end{frame}

\begin{frame}{Scheme at a glance}
  \begin{itemize}
    \item Cell-centered Cartesian grid, two ghost layers~\cite{versteeg2007,moukalled2016}.
    \item MUSCL on primitives with minmod (MC, van Leer optional)~\cite{toro2009}.
    \item HLLD Riemann solver~\cite{miyoshi2005}; reduces to HLLC for $B = 0$.
    \item GLM $\psi$/$B_x$ coupling baked into the Riemann solver; parabolic damping per RK stage~\cite{dedner2002}.
    \item SSP-RK2 time integration~\cite{shu1988}, CFL $= 0.3$.
  \end{itemize}
\end{frame}

\begin{frame}{HLLD wave pattern}
    \begin{itemize}
      \item Five-wave approximation to the full seven-wave MHD fan~\cite{miyoshi2005}.
      \item Bounding speeds $S_L, S_R$ from Davis-Einfeldt; entropy speed $S_M$ from jump conditions.
      \item Two starred + two double-starred states across the rotational Alfven discontinuities.
      \item Positivity-preserving for physical inputs.
    \end{itemize}

\section{Discussion of Results}

\begin{frame}{Accuracy vs limiter choice}
  \begin{itemize}
    \item At $128^{2}$ here: $\rho_{\min}\approx 0.47$, $\rho_{\max}\approx 2.16$.
    \item Athena++ with MC at the same resolution: $\rho_{\min}\sim 0.1$, $\rho_{\max}\sim 3.5$~\cite{athena_orszag_tang}.
    \item Gap is the limiter: minmod is the most diffusive TVD choice and smears current sheets~\cite{toro2009}.
    \item MC or van Leer would close it---but cost robustness in low-beta cells.
  \end{itemize}
\end{frame}

\begin{frame}{Why between first and second order}
  \begin{itemize}
    \item MUSCL is second-order in smooth regions~\cite{toro2009}.
    \item TVD limiters drop to first order at extrema and at shocks.
    \item Orszag-Tang at $t = 0.5$ is shock-dominated $\Rightarrow$ global rate between 1 and 2.
    \item Observed: $\approx 0.8$ coarse (shocks dominate), $\approx 1.3$ fine (smooth regions start to dominate).
  \end{itemize}
\end{frame}

\begin{frame}{Limitations}
  \begin{itemize}
    \item Vectorized NumPy: $128^{2}$ in $\sim 7$\,s; $256^{2}$ is the practical ceiling on a laptop.
    \item No constrained-transport option---GLM is simpler but less strict on $\nabla\cdot B$~\cite{balsara1999}.
    \item 2D Cartesian; no AMR.
    \item Linear reconstruction (MUSCL); no WENO, no higher-order time integration.
  \end{itemize}
\end{frame}

\section{Conclusions}

\begin{frame}{Conclusions}
  \begin{itemize}
    \item 2D ideal MHD solver from scratch: Godunov FVM, MUSCL-minmod, HLLD~\cite{miyoshi2005}, GLM~\cite{dedner2002}, SSP-RK2~\cite{shu1988}.
    \item Orszag-Tang at $128^{2}$ reproduces the canonical $t = 0.5$ density/pressure pattern~\cite{londrillo2000,athena_orszag_tang}.
    \item Mass, momentum, energy conserved to machine precision; GLM bounds $\max|\nabla\cdot B|$.
    \item Convergence rate between 1 and 2, as expected for MUSCL-minmod on shock-dominated flow.
  \end{itemize}
  \vspace{0.5em}
  \textbf{Natural extensions}: constrained transport, resistive MHD, 3D, AMR, Numba/JAX.
\end{frame}
